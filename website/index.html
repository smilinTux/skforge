# Web Server Blueprint category: web-server language: rust framework: axum features: - static-files - json-api - middleware - graceful-shutdown config: port: 8080 host: 0.0.0.0 max-connections: 1000
// Generated Rust Web Server use axum::{ extract::Query, http::StatusCode, response::Json, routing::{get, post}, Router, }; use std::net::SocketAddr; use tokio::signal; use tower_http::services::ServeDir; async fn main() { let app = Router::new() .route("/", get(|| async { "Hello, World!" })) .route("/api/health", get(health_check)) .nest_service("/static", ServeDir::new("assets")); let addr = SocketAddr::from(([0, 0, 0, 0], 8080)); let listener = tokio::net::TcpListener::bind(&addr) .await.unwrap(); axum::serve(listener, app) .with_graceful_shutdown(shutdown_signal()) .await.unwrap(); }
